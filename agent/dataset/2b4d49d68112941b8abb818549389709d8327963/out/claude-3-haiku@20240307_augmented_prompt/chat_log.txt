    private static final com.hazelcast.cluster.Cluster cluster = hazelcastInstance.getCluster();

     * @param <K> the type of the keys stored in the cache
     * @param <V> the type of the values stored in the cache
     */

        IMap<K, V> map = hazelcastInstance.getMap(name);

        ClusteredCache<K, V> cache = new ClusteredCache<>(map, timeToLive, timeUnit);
        caches.put(name, cache);
        return cache;
    }

     * @param <K> the type of the keys stored in the cache
     * @param <V> the type of the values stored in the cache
     * @param name the name of the cache
     * @return the clustered cache
     */
    @SuppressWarnings("unchecked")
    public static <K, V> ClusteredCache<K, V> getCache(String name) {
        return (ClusteredCache<K, V>) caches.get(name);
    }

    /**
     * Removes a clustered cache.
     * @param name the name of the cache
     */
    public static void removeCache(String name) {
        caches.remove(name);
    }
